⚙️ CONFIGURATION (.env)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Current settings in .env:

  # Sync Configuration
  REALTIME_SYNC=true                 # Enable/disable sync
  SYNC_DIRECTION=sqlite_to_mysql     # (sqlite_to_mysql | mysql_to_sqlite | both)

  # MySQL Configuration (XAMPP)
  MYSQL_HOST=127.0.0.1
  MYSQL_PORT=3306
  MYSQL_USER=root
  MYSQL_PASSWORD=                    # (empty for XAMPP default)
  MYSQL_DB=epi_detection_db

  # Database Type
  DB_TYPE=sqlite                     # Primary DB type

To change sync direction at runtime:
  sqlite_to_mysql  → Primary: SQLite, Replicated to MySQL
  mysql_to_sqlite  → Primary: MySQL, Replicated to SQLite  
  both             → Write to both simultaneously (slower, guaranteed sync)

🔄 HOW IT WORKS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

EXAMPLE: Using the app normally

  # Your Flask code - NO CHANGES NEEDED
  detection = Detection(image_path='/path', total_persons=5, ...)
  db.session.add(detection)
  db.session.commit()

  BEHIND THE SCENES:
  1. Wrote to PRIMARY DB (SQLite)
  2. Sync hook intercepted the INSERT
  3. Queued operation for replication
  4. Replicated to SECONDARY DB (MySQL) ← AUTO!
  5. Both DBs now in sync
